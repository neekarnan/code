Plot 1D signals (of your choice) for each of the following cases
Case 1
Independent variable: Continuous
Dependent variable: Continuous

import numpy as np
import matplotlib.pyplot as plt
t = np.linspace(0, 2*np.pi, 500)  # 500 points between 0 and 2pi
y = np.sin(t)  # continuous dependent variable
plt.figure(figsize=(10,4))
plt.plot(t, y, color='blue')
plt.title("Continuous x → Continuous y (Sine wave)")
plt.xlabel("Time")
plt.ylabel("Amplitude")
plt.grid(True)
plt.show()

Independent variable: Continuous
Dependent variable: Discrete

y_discrete = np.round(np.sin(t))  # discretized y
plt.figure(figsize=(10,4))
plt.plot(t, y_discrete, drawstyle='steps-mid', color='red')
plt.title("Continuous x → Discrete y (Rounded Sine)")
plt.xlabel("Time")
plt.ylabel("Amplitude (Discrete)")
plt.grid(True)
plt.show()

import numpy as np
import matplotlib.pyplot as plt

x = np.linspace(-5, 5, 100)
y = np.linspace(-5, 5, 100)

X, Y = np.meshgrid(x, y)

Z_levels = [0, 1, 2]

fig = plt.figure(figsize=(8,6))
ax = fig.add_subplot(111, projection='3d')

for z in Z_levels:
    ax.plot_surface(
        X, Y, z * np.ones_like(X),
        alpha=0.8
    )

ax.set_title("Continuous X, Y → Discrete Z (Air Quality Levels)")
ax.set_xlabel("X (Continuous Location)")
ax.set_ylabel("Y (Continuous Location)")
ax.set_zlabel("Z (Discrete AQI Levels)")
plt.show()

import numpy as np
import matplotlib.pyplot as plt

time = np.linspace(0, 24, 200)  # 24 hours
stations = np.array([0, 1, 2, 3, 4])

fig = plt.figure(figsize=(8,6))
ax = fig.add_subplot(111, projection='3d')

for station in stations:
    temperature = 20 + 5*np.sin(2*np.pi*time/24) + station

    ax.plot(time, station*np.ones_like(time), temperature, linewidth=2)

ax.set_title("Temperature over Time at Discrete Weather Stations")
ax.set_xlabel("Time (hours)")
ax.set_ylabel("Weather Station (Discrete)")
ax.set_zlabel("Temperature (°C)")
plt.show()

Dependent variable: Discrete

import numpy as np
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import Axes3D

servers = np.arange(1, 4)
time = np.linspace(0, 12, 200)

fig = plt.figure(figsize=(10,7))
ax = fig.add_subplot(111, projection='3d')

for s in servers:

    activity = np.ones_like(time)

    if s == 1:
        activity[(time >= 0) & (time <= 1)] = 2
    elif s == 2:
        activity[(time >= 5) & (time <= 9)] = 2
    elif s == 3:
        activity[(time >= 2) & (time <= 4)] = 2

    ax.plot(
        np.full_like(time, s),
        time,
        activity,
        linewidth=2,
        label=f"Server {s}",
        drawstyle='steps-post'
    )

ax.set_xlabel("Server (discrete)")
ax.set_ylabel("Time (hours, continuous)")
ax.set_zlabel("Activity Level (discrete)")
ax.set_title("Discrete – Continuous – Discrete: Server Activity Level")
ax.legend()
plt.show()
