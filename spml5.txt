# Recording Audio Sample

import sounddevice as sd
from scipy.io.wavfile import write
import numpy as np
print("Recording... Please speak now")
fs = 44100   # Sampling frequency
duration = 30   # seconds
recording = sd.rec(int(duration * fs), samplerate=fs, channels=1)
sd.wait()
print("Recording finished")
write("original.wav", fs, recording)

# Reading the Recorded Audio File
from scipy.io import wavfile
fs, data = wavfile.read("original.wav")
data = data.astype(np.float32)   # Convert to float for processing
N = len(data)
print("Number of samples (N):", N)

# 3-Sample Sliding Window Averaging

window_size = 3
averaged_data = np.zeros(N - window_size + 1)

for i in range(N - window_size + 1):
    averaged_data[i] = np.mean(data[i:i + window_size])

# Normalize before saving

averaged_data_int = np.int16(
    averaged_data / np.max(np.abs(averaged_data)) * 32767
)

write("averaged.wav", fs, averaged_data_int)

# Play the audio files

from IPython.display import Audio
