Qn:1 Generate a random sequence of numbers for n = {0, 1, 2, 3, ....., 100}

import numpy as np
import matplotlib.pyplot as plt
n = np.arange(0, 101)
x = np.random.rand(len(n))
h = np.array([4, 3, 2, 1])
y = np.convolve(x, h)
plt.figure(figsize=(8, 8))
plt.subplot(3, 1, 1)
plt.stem(n, x, use_line_collection=True)
plt.title("Input Sequence x[n]")
plt.xlabel("n")
plt.ylabel("x[n]")

plt.subplot(3, 1, 2)
plt.stem(range(len(h)), h, use_line_collection=True)
plt.title("Impulse Response h[n]")
plt.xlabel("n")
plt.ylabel("h[n]")

plt.subplot(3, 1, 3)
plt.stem(range(len(y)), y, use_line_collection=True)
plt.title("Output Sequence y[n]")
plt.xlabel("n")
plt.ylabel("y[n]")

plt.tight_layout()
plt.show()

2.To perform 2D convolution
the effect on the image. on a digital image using a 3 x 3averaging filter and observe
Given
Input image: Selfie image
Image representation: Digital image read as a matrix
Only one colour plane is considered (grayscale)

import cv2
import numpy as np
import matplotlib.pyplot as plt
img = cv2.imread("selfie.jpg")
gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
h = np.ones((3, 3)) / 9
output = cv2.filter2D(gray, -1, h)
plt.figure(figsize=(10, 4))
plt.subplot(1, 2, 1)
plt.title("Original Image")
plt.imshow(gray, cmap='gray')
plt.axis('off')
plt.subplot(1, 2, 2)
plt.title("Output Image after 2D Convolution")
plt.imshow(output, cmap='gray')
plt.axis('off')
plt.show()
